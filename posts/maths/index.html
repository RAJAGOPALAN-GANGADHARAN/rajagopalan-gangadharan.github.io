Useful nCr Technique

Divide by their GCD’s to prevent overflow

For fast Calculation:

ll NcR(ll n, ll r) 
{  
    ll p = 1, k = 1;  
    if (n - r < r) 
        r = n - r; 

    if (r != 0) { 
        while (r) { 
            p *= n; 
            k *= r; 
            ll m = __gcd(p, k); 
            p /= m; 
            k /= m; 
            n--; 
            r--; 
        } 
    } 

    else
        p = 1; 

    return p;
}
